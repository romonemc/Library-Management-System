InsertAt

IF index < 0 THEN
	RETURN NULL
END IF

INTEGER current = 1
Book* currNode = head

WHILE currNode AND index > current DO
	currNode = currNode->nextBook
	current = current + 1
END WHILE

IF index > 0 AND currNode = NULL THEN
	RETURN NULL
END IF

Book* newBook = new Book(b->isbn, b->quantity, b->edition, b->name, b->category, b->publisher, b->authors)

IF index = 0 THEN
	try
		newBook->nextBook = head;
		head = newBook;
		WRITE "Book inserted successfully!"
	catch
		WRITE "Oops :( seems something went wrong. No harm, no foul just try again."
ELSE
	try
		newBook->nextBook = currNode->nextBook
		currNode->nextBook = newBook;
		WRITE "Book inserted successfully!"
	catch
		WRITE "Oops :( seems something went wrong. No harm, no foul just try again."
END IF

RETURN 0

Count

INTEGER count = 0
Book* currBook = head;

WHILE currBook <> NULL
	currBook = currBook->getNextBook()
	count = count + 1
END WHILE

RETURN count

Traversing the list

INTEGER count = 0
Book* currBook = head;

WHILE currBook <> NULL
	WRITE currBook->isbn
	WRITE currBook->name
